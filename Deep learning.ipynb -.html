def make_gradcam_heatmap(img_array, model, last_conv_layer_name, pred_index=None):
    # Ensure img_array has a batch dimension and is float32
    if img_array.ndim == 3:
        img_array = np.expand_dims(img_array, axis=0)
    img_array = tf.cast(img_array, dtype=tf.float32)

    # First, we create a model that extracts the activations from the last conv layer
    conv_layer = model.get_layer(last_conv_layer_name)
    grad_model = tf.keras.models.Model(model.inputs, conv_layer.output)

    # Second, we create a model that maps the activations of the last conv layer
    # to the final class predictions
    classifier_input = tf.keras.Input(shape=conv_layer.output.shape[1:])
    x = classifier_input

    found_conv_layer = False # Initialize the flag
    # Iterate through layers after the last_conv_layer_name to build the classifier part
    # This assumes the model is Sequential or has a clear flow after the conv layer
    for layer in model.layers:
        if layer.name == last_conv_layer_name:
            found_conv_layer = True
            continue
        if found_conv_layer: # Process only layers after the target conv layer
            x = layer(x)

    classifier_model = tf.keras.models.Model(classifier_input, x)

    # Record operations for automatic differentiation
    with tf.GradientTape() as tape:
        # Explicitly watch the input tensor. This ensures gradients are computed
        tape.watch(img_array)

        # Get the activations of the last conv layer
        last_conv_layer_output = grad_model(img_array)

        # Pass the activations through the classifier part of the model
        preds = classifier_model(last_conv_layer_output)

        if pred_index is None:
            pred_index = tf.argmax(preds[0])
        class_channel = preds[:, pred_index]

    # Gradients of the top predicted class with respect to the output of the last conv layer
    grads = tape.gradient(class_channel, last_conv_layer_output)

    # This is a vector where each entry is the mean intensity of the gradient
    # over a specific feature map channel
    if grads is None:
        print("Warning: Gradients could not be computed. Returning empty heatmap.")
        return np.zeros(last_conv_layer_output.shape[1:3]) # Return an empty heatmap of appropriate size

    pooled_grads = tf.reduce_mean(grads, axis=(0, 1, 2))

    # Multiply each channel in the feature map array by "how important" this channel is
    # with respect to the top predicted class
    last_conv_layer_output = last_conv_layer_output[0] # Get the output for the single image
    heatmap = last_conv_layer_output @ pooled_grads[..., tf.newaxis]
    heatmap = tf.squeeze(heatmap)

    # For visualization purpose, we will clip the values below 0 to 0.
    # Then normalize the heatmap to be between 0 and 1.
    heatmap = tf.maximum(heatmap, 0) / tf.math.reduce_max(heatmap)
    return heatmap.numpy()

def display_gradcam(img_path, heatmap, alpha=0.4):
    # Muat gambar aslinya
    img = cv2.imread(img_path)
    img = cv2.resize(img, (224, 224)) # Pastikan ukuran sama dengan input model

    # Resize heatmap ke ukuran gambar asli
    heatmap = cv2.resize(heatmap, (img.shape[1], img.shape[0]))

    # Konversi heatmap dari float [0, 1] ke byte [0, 255]
    heatmap = np.uint8(255 * heatmap)

    # Gunakan skema warna JET untuk menghasilkan peta panas berwarna
    jet = plt.colormaps['jet'] # Ganti cm.jet dengan plt.colormaps['jet']

    # Gunakan nilai RGB dari skema warna JET
    jet_colors = jet(np.arange(256))
    jet_heatmap = jet_colors[heatmap]

    # Konversi ke RGB
    jet_heatmap = (jet_heatmap[:, :, :3] * 255).astype(np.uint8)

    # Gabungkan peta panas dengan gambar asli
    superimposed_img = jet_heatmap * alpha + img
    superimposed_img = np.clip(superimposed_img, 0, 255).astype(np.uint8)

    return superimposed_img
